USERS

    Post new user:
        POST users/register
        example body:
            {
                "username": "Charlie",
                "password": "CharliesPassword"
            }

    Get user by username:
        GET users/<username>

    Delete user by username:
        DELETE users/<username>

    Login user:
        POST users/login
        example body:
            {
                "username": "Charlie",
                "password": "CharliesPassword"
            }

TRIPS

    Create trip:
        POST trips/
        example body:
        {
            "tripName" : "YosemiteTrip",
            "ownerId": "fe63edb8-7a17-4834-979b-bf0991310f2f",
            "description": "We're going to Yosemite!",
            "recArea": "2991"
        }

    Create trip date:
        POST trips/:tripId
        example body:
        {
            "date": "2025-07-03T00:00:00Z"
        }

    Get trips by username:
        GET trips/user/:username

RIDB

    get rec areas by query:
        GET ridb/recareas?query=<searchterm>

        optional limit query param, default 50
        GET ridb/recareas?query=<searchterm>&limit=<number>
